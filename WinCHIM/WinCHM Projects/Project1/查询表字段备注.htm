select b.表名,b.表说明,'''''as '||b.列名||',--'||replace(b.备注,chr(10),'')||b.类型,b.类型,b.长度,replace(b.备注,chr(10),'')
from
(select a.TABLE_NAME as 表名,
c.comments as 表说明,
a.column_name as 列名,
a.data_type as 类型,
a.data_length as 长度 ,
b.comments as 备注
from （select
lower(table_name) as TABLE_NAME ,
              data_type as data_type ,
              column_name as column_name,
              data_length as data_length from user_tab_cols）a
left join (select lower(table_name) as table_name ,
            column_name as column_name ,
            comments as comments from user_col_comments) b           
on a.COLUMN_NAME = b.column_name and a.TABLE_NAME = b.table_name
left join user_tab_comments c on a.TABLE_NAME = lower(c.table_name)
--where a.TABLE_NAME = 'iem_basicinfoxy_default'
union all
select a.table_name as 表名,
c.comments as 表说明,
a.column_name as 列名,
a.data_type as 类型,
a.data_length as 长度 ,
b.comments as 备注
from user_tab_cols a
left join user_col_comments b on a.COLUMN_NAME = b.column_name and a.TABLE_NAME = b.table_name
left join user_tab_comments c on a.TABLE_NAME = c.table_name) b
where b.表名 = 'hmis_doctor_adviceinfo';
