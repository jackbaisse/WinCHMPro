--------查询表字段-------
select a.TABLE_NAME||'_'||a.COLUMN_NAME as 主键, a.TABLE_NAME as 表名,a.COLUMN_NAME as 列名,
a.DATA_TYPE as 数据类型,a.DATA_LENGTH as 宽度,a.COLUMN_ID as 列排序
from  user_tab_cols a
left  join user_tab_comments b on a.TABLE_NAME=b.TABLE_NAME
where b.table_type='TABLE';
