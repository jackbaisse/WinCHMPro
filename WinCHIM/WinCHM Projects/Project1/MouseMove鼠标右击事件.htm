private void gridControl2_MouseMove(object sender, MouseEventArgs e)
        {
            ContextMenuStrip a = gridControlLeft.ContextMenuStrip;
            foreach (ToolStripMenuItem item in a.Items)
            {
                if (item.Text == "交班报告") item.Visible = true;
                else if (item.Text == "全选") item.Visible = true;
                else if (item.Text == "取消全选") item.Visible = true;
                else if (item.Text == "删除") item.Visible = false;
                else if (item.Text == "保存") item.Visible = false;
            }
        }
