select t.segment_name, t.segment_type, sum(t.bytes / 1024 / 1024) "占用空间(M)"
,a.TABLESPACE_NAME from dba_segments t
left join USER_TABLES a on a.TABLE_NAME = t.segment_name
where t.segment_type='TABLE' and t.bytes >(10*1024*1024）--大于10M的文件
group by t.segment_type,OWNER,  t.segment_name ,a.TABLESPACE_NAME
