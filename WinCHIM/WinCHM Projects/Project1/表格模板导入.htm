select b.* from commonnote_item a
left join dateelement b on b.elementflow=a.dataelementflow and b.elementid=a.dataelementid
left join commonnote c on a.commonnoteflow=c.commonnoteflow
left join commonnote_tab d on a.commonnoteflow=d.commonnoteflow and a.commonnote_tab_flow=d.commonnote_tab_flow
where  c.commonnoteflow ='bf035b9e-cf90-434f-b40f-51f70ca78782'
