1、多行转一列

SELECT  p1icbz ,
        value = ( STUFF(( SELECT    ',' + p1icno
                          FROM      pz01d
                          where p1icbz=Test.p1icbz
                        FOR
                          XML PATH('')
                        ), 1, 1, '') )
FROM    pz01d AS Test
GROUP BY p1icbz ;
