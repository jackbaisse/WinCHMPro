create or replace procedure yg_bgcard_update
as
declare @zyh varchar2(20)
        set @zyh ='select b.noofinpat from yg_bgcard a ,inpatient b where a.crbid=b.patnoofhis'
        select @zyh
begin
    if
      update yg_bgcard a set noofinpat=@zyh
    else
      update bgcard a set iscomplete='1' where noofinpat=(select noofinpat from yg_bgcard b where a.noofinpat=b.noofinpat)
     -- update yg_bgcard a set noofinpat=(select noofinpat from inpatient b where a.crbid=b.patnoofhis)
  end;
